1. Crie um código em C para escrever "Ola mundo!" em um arquivo chamado 'ola_mundo.txt'.

#include <stdio.h>
#include <stdlib.h>

int main (int argc, const char *argv[]){

FILE *fp;
char string[100];
int i;
fp = fopen ("ola_mundo.txt", "w");

gets(string);

for (i = 0; string[i]; i++){
	putc(string[i], fp);

}

fclose(fp);

return 0;
}
